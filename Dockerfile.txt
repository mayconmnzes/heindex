# ESTÁGIO 1: Compilação (Build)
# Usa uma imagem do Maven com JDK 17 para gerar o arquivo .jar
FROM maven:3.9.6-eclipse-temurin-17 AS build
WORKDIR /app

# Copia os arquivos de configuração do Maven e o código fonte
COPY pom.xml .
COPY src ./src

# Executa o comando para gerar o JAR, pulando os testes para agilizar o deploy
RUN mvn clean package -DskipTests

# ESTÁGIO 2: Execução (Runtime)
# Usa uma imagem leve do JDK para rodar a aplicação
FROM eclipse-temurin:17-jdk-jammy
WORKDIR /app

# Copia apenas o arquivo JAR gerado no estágio anterior
COPY --from=build /app/target/*.jar app.jar

# Define a porta que o Render costuma usar
EXPOSE 8080

# Comando para iniciar a aplicação
ENTRYPOINT ["java", "-jar", "app.jar"]
